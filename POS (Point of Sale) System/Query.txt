//Create pos_tbl
CREATE TABLE POS_TB (
    TRANSID      VARCHAR2(12 BYTE) PRIMARY KEY,
    CUSTOMERNAME VARCHAR2(50 BYTE) NOT NULL,
    ITEMNAME     VARCHAR2(50 BYTE) NOT NULL,
    TRANS_DATE   DATE NOT NULL,
    QUANTITY     NUMBER,
    PRICE        NUMBER,
    TOTALAMOUNT  NUMBER,
    REMARKS      VARCHAR2(100 BYTE)
);

CREATE SEQUENCE POS_SEQ
START WITH 10
MINVALUE 10
MAXVALUE 99
INCREMENT BY 1;
